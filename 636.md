class Solution:
    def exclusiveTime(self, n, logs):
        """
        :type n: int
        :type logs: List[str]
        :rtype: List[int]
        """
        true = [0 for i in range(n)]
        stack = []
        last_time = 0
        for log in logs:
            op = log.split(':')
            if op[1] == 'start':
                if stack != []:
                    true[stack[-1]] += int(op[2]) - last_time
                last_time = int(op[2])
                stack.append(int(op[0]))
            else:
                c = stack.pop()
                true[c] += int(op[2]) - last_time + 1
                last_time = int(op[2]) + 1
        return true
